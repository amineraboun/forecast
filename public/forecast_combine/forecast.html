 197    def split_procedure_summary(self, verbose: bool=True) -> dict:
 198        """
 199        Generate a summary of the cross-validation procedure.
 200
 201        Parameters:
 202        -----------        
 203            verbose : bool, optional
 204                If True, print the summary. Default is True.
 205
 206        Returns:
 207        --------        
 208            dict
 209                A dictionary containing the summary of the cross-validation procedure.
 210        """
 211
 212        _n_splits = self._cv.get_n_splits(self._y)
 213        cutoffs = [self._y.index[_train[-1]] for (_train, _) in self._cv.split(self._y.index)]
 214        _split_proc= {'Number of Folds': _n_splits,
 215                      'Initial Window Size': self._cv.initial_window,
 216                      'Step Length': self._cv.step_length,
 217                      'Forecast Horizon': len(self._cv.fh),
 218                      'First Cutoff': cutoffs[0],
 219                      'Last Curoff': cutoffs[-1]
 220                     }
 221        if verbose:
 222            for k, v in _split_proc.items():
 223                print(f"{k:<21}: {v}")
 224        return _split_proc

 277    def predict(self, 
 278                X: Optional[pd.DataFrame] = None, 
 279                fh: Optional[ForecastingHorizon] = None, 
 280                coverage: float = 0.9,
 281                verbose = False
 282               ) -> Tuple[pd.DataFrame, pd.DataFrame]:
 283        """
 284        Generate predictions (Average and confidence interval) using the fitted model.
 285
 286        Parameters:
 287        -----------        
 288            X : pd.DataFrame, optional
 289                Exogenous variables for forecasting. Default is None
 290            fh : ForecastingHorizon, optional
 291                Forecast horizon. Default is None and takes the horizon defined at instantiation.
 292            coverage : float, optional
 293                The coverage of the confidence interval. Default is 0.9.
 294
 295        Returns:
 296        --------        
 297            Tuple[pd.DataFrame, pd.DataFrame]:
 298                A tuple containing the predictions and the confidence intervals.
 299        """
 300        if self.is_fitted==False:
 301            if verbose:
 302                print(f"\n{self.forecaster_name} model not fitted yet, or fitted on a subset only")
 303                print("Fitting the model on the entire sample ...")
 304            self.fit(on='all', fh=fh)
 305
 306        if fh is None:
 307            fh =  self._fh
 308        if X is None:
 309            X = self._X
 310
 311        y_pred = self.forecaster.predict(X=X, fh=fh)
 312        try:
 313            y_pred_ints = self.forecaster.predict_interval(X=X, fh=fh, coverage=coverage)
 314        except Exception as e:
 315            if verbose:
 316                print(f"{self.forecaster_name} does not support prediction intervals")
 317                print(f"Error: {e}")
 318                print("Computing the prediction intervals based on historical errors distribution")
 319            
 320            historical_errors = self.get_pred_errors()
 321            y_pred_ints = calculate_prediction_interval(historical_errors, y_pred, coverage=coverage)
 322
 323        return y_pred, y_pred_ints

 325    def update(self, 
 326               new_y: pd.Series,
 327               new_X: Optional[pd.DataFrame] = None,
 328               fh: Optional[ForecastingHorizon] = None, 
 329               coverage: float = 0.9, 
 330               refit: bool = False,
 331              ) -> Tuple[pd.DataFrame, pd.DataFrame]:
 332        """
 333        Update cutoff value to forecast new dates.
 334        Possibility to refit the model.
 335
 336        Parameters:
 337        -----------        
 338            new_y : pd.Series
 339                The new dependent variable values.
 340            new_X : pd.DataFrame, optional
 341                The new exogenous variables. Default is None.
 342            fh : ForecastingHorizon, optional
 343                Forecast horizon. Default is None and takes the horizon defined at instantiation.
 344            coverage : float, optional
 345                The coverage of the confidence interval. Default is 0.9.
 346            refit : bool, optional
 347                If True, refit the model. Default is False.
 348
 349        Returns:
 350        --------        
 351            Tuple[pd.DataFrame, pd.DataFrame]:
 352                A tuple containing the updated predictions and the updated confidence intervals.
 353        """
 354        new_y = new_y.resample(self.freq).last().ffill()
 355        if new_X is not None:
 356            new_X = new_X.resample(self.freq).last().ffill()
 357        self.forecaster.update(y=new_y, X=new_X, update_params=refit)
 358        self._y = new_y
 359        self._X = new_X
 360        y_pred, y_pred_ints = self.predict(X=new_X, fh=fh, coverage=coverage)
 361        return y_pred, y_pred_ints

 460    def __clean_data(self, 
 461                     data: pd.DataFrame, 
 462                     depvar_str: str, 
 463                     exog_l: Optional[list],
 464                     freq: str
 465                    ) -> Tuple[pd.Series, pd.DataFrame]:
 466        """
 467        Reformat the data taking into account the requested frequency.
 468        """
 469
 470        _df = data.dropna().resample(freq).last().copy()
 471        _df.index.freq = freq
 472        # Declare and stage the variable to forecast
 473        y = _df[depvar_str]
 474        
 475
 476        # List of Exogenous variables if any
 477        if exog_l is None:
 478            X = None
 479        else:
 480            assert all([c in _df.columns for c in exog_l]), 'not all columns are not in the data'
 481            X = _df[exog_l]
 482        return y, X

 792        """
 793        Compute the insample predictions for the fitted model.
 794
 795        Parameters:
 796        -----------

 798                The number of samples to compute. Default is 100.
 799

 801        --------        
 802            pd.DataFrame
 803                A DataFrame containing the insample predictions.
 804        """
 805
 806        if self.is_fitted:
 807            _y = self._y
 808        else:
 809            _y = self._y_train
 810
 811        insample_eval_window = len(_y) - len(self._fh)
 812        if random_sample:            
 813            nsample = max(insample_eval_window, nsample)
 814            # Randomly selecting cutoff points
 815            _cutoffs = np.random.choice(insample_eval_window, size=nsample, replace=False)
 816            cv_in = CutoffSplitter(cutoffs=_cutoffs, window_length=1, fh=self._fh)
 817        else:
 818            _cutoffs = np.arange(insample_eval_window)
 819            cv_in = ExpandingWindowSplitter(initial_window=1, step_length=1, fh=self._fh)
 820        if verbose:
 821            print(f"\nComputing {self.forecaster_name} forecaster historic predictions....")        
 822
 823        params = [(self.forecaster, self._X, intrain, intest, verbose) for intrain, intest in cv_in.split_series(_y)]                
 824        with Pool() as pool:
 825            insample_result = list(tqdm(pool.imap(compute_predictions, params), total=len(params)))
 826
 827        insample_result_df = pd.concat(insample_result)
 828        if insample_result_df.empty:
 829            print(f"No insample predictions computed {self.forecaster_name}")
 830        if verbose:
 831            print(f"\n{self.forecaster_name} forecaster historic predictions completed")

 833        return insample_result_df
